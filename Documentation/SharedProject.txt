  CREATE netstandard2.0 PROJECT
dotnet new classlib -n Revit.Common -f netstandard2.0

  REMOVE UNNECESSARY FILES
cd Revit.Common;
Remove-Item -Path "Revit.Common.csproj";
Remove-Item -Path "Class1.cs";

  CONVERT PROJECT INTO A SHARED PROJECT
cd Revit.Common;
New-Item -Path "Revit.Common.shproj";
New-Item -Path "Revit.Common.projitems";

  SET THE MINIMAL .shproj CONFIGURATION
<?xml version="1.0" encoding="utf-8"?>
<Project ToolsVersion="15.0" xmlns="http://schemas.microsoft.com/developer/msbuild/2003">
  <PropertyGroup>
    <ProjectGuid>bb650bcb-b8d2-4400-aa10-03a8c66b878c</ProjectGuid>
  </PropertyGroup>
  <Import Project="Revit.Common.projitems" Label="Shared" />
</Project>

  ADD THE VISUAL STUDIO ENABLING DIRECTIVES
  THIS ENABLES THE LOADING OF THE PROJECT INTO VISUAL STUDIO
<?xml version="1.0" encoding="utf-8"?>
<Project ToolsVersion="15.0" xmlns="http://schemas.microsoft.com/developer/msbuild/2003">
  <PropertyGroup>
    <ProjectGuid>bb650bcb-b8d2-4400-aa10-03a8c66b878c</ProjectGuid>
  </PropertyGroup>
  <Import Project="Revit.Common.projitems" Label="Shared" />

  <Import Project="$(MSBuildExtensionsPath)\$(MSBuildToolsVersion)\Microsoft.Common.props"
    Condition="Exists('$(MSBuildExtensionsPath)\$(MSBuildToolsVersion)\Microsoft.Common.props')" />
  <Import
    Project="$(MSBuildExtensionsPath32)\Microsoft\VisualStudio\v$(VisualStudioVersion)\CodeSharing\Microsoft.CodeSharing.Common.Default.props" />
  <Import
    Project="$(MSBuildExtensionsPath32)\Microsoft\VisualStudio\v$(VisualStudioVersion)\CodeSharing\Microsoft.CodeSharing.Common.props" />
  <PropertyGroup />
  <Import
    Project="$(MSBuildExtensionsPath32)\Microsoft\VisualStudio\v$(VisualStudioVersion)\CodeSharing\Microsoft.CodeSharing.CSharp.targets" />

</Project>

  SET THE MINIMAL .projitems CONFIGURATION
<?xml version="1.0" encoding="utf-8"?>
<Project xmlns="http://schemas.microsoft.com/developer/msbuild/2003">
  <PropertyGroup>
    <HasSharedItems>true</HasSharedItems>
    <SharedGUID>e11fb2b1-2af0-46cc-9302-9b3579a48115</SharedGUID>
  </PropertyGroup>
  <PropertyGroup Label="Configuration">
    <Import_RootNamespace>Common</Import_RootNamespace>
  </PropertyGroup>
</Project>
